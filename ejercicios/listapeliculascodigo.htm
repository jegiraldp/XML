<!DOCTYPE HTML PUBLIC "-//IETF//DTD HTML//EN">

<html>
<head>
<title>Enumerar Peliculas</title>
<script language="JavaScript">
var xmldoc;

function LoadDoc() {
xmldoc = new ActiveXObject("microsoft.xmldom");
xmldoc.load("peliculas.xml");
}

function ListaPeliculas() {
LoadDoc();
root = xmldoc.documentElement;
for (i = 0; i < root.childNodes.length; i++) {
DOCCONTENT.innerHTML += "<b>" +
root.childNodes.item(i).firstChild.text + "</b><br>";
DOCCONTENT.innerHTML += root.childNodes.item(i).lastChild.text + "<br><br>";
}
}
</script>
</head>

<body>
Haga clic en el boton para enumerar las peliculas del documento XML Peliculas.
<br>
<input type="button" value="Enumerar peliculas" onclick="ListaPeliculas()">
<div id="DOCCONTENT"></div>
</body>
